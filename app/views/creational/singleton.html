Description: create single instance of a class

Scenario: only want one reference to a class

Demo: very simple counter class

var singleton = (function () {
  var instance = null;

  function Counter () {
    var count = 0;

    return {
      incr: function () {
        return ++count;
      },

      getCount: function () {
        return count;
      }
    };
  }

  return {
    getInstance: function () {
      if (!instance) {
        instance = Counter();
      }

      return instance;
    }
  };
})();

var a = singleton.getInstance();
a.incr();
console.log(a.getCount()); // 1
a.incr();
console.log(a.getCount()); // 2

var b = singleton.getInstance();
console.log(b.getCount()); // 2
b.incr();

console.log(a.getCount()); // 3
